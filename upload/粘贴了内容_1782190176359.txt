    for key in "${!MAP[@]}"; do
        placeholder="{{${key}}}"
        value="${MAP[$key]}"
        sed -i "s|${placeholder}|${value}|g" "$file"
    done
}

import sys
import os
from pathlib import Path

os.environ['PYTHONIOENCODING'] = 'utf-8'

# 文件读取封装
def safe_read(file_path, encoding='utf-8'):
    """安全读取文件，自动处理编码"""
    try:
        with open(file_path, 'r', encoding=encoding) as f:
            return f.read()
    except UnicodeDecodeError:
        # 回退到系统默认编码
        with open(file_path, 'r', encoding='gbk') as f:
            return f.read()
    except Exception as e:
        raise RuntimeError(f"文件读取失败: {e}")

def safe_write(file_path, content, encoding='utf-8'):
    """安全写入文件"""
    with open(file_path, 'w', encoding=encoding, newline='') as f:
        f.write(content)

# 配置文件读取
def load_yaml(file_path):
    """加载YAML配置"""
    import yaml
    content = safe_read(file_path)
    return yaml.safe_load(content)

# CSV文件处理
import pandas as pd
def safe_read_csv(file_path):
    """安全读取CSV"""
    try:
        return pd.read_csv(file_path, encoding='utf-8')
    except UnicodeDecodeError:
        return pd.read_csv(file_path, encoding='gbk')

# process_utils.py
import platform
import subprocess
import os
import signal

class ProcessManager:
    """跨平台进程管理"""
    
    @staticmethod
    def is_windows():
        return platform.system() == 'Windows'
    
    @staticmethod
    def kill_process_by_name(pattern):
        """根据进程名杀死进程"""
        if ProcessManager.is_windows():
            # Windows: 使用taskkill
            try:
                subprocess.run(
                    ['taskkill', '/F', '/IM', pattern],
                    capture_output=True,
                    text=True,
                    timeout=10
                )
            except subprocess.TimeoutExpired:
                pass
        else:
            # Linux: 使用pkill
            try:
                subprocess.run(
                    ['pkill', '-f', pattern],
                    capture_output=True,
                    timeout=10
                )
            except subprocess.TimeoutExpired:
                pass
    
    @staticmethod
    def kill_process_by_pid(pid):
        """根据PID杀死进程"""
        if ProcessManager.is_windows():
            subprocess.run(['taskkill', '/F', '/PID', str(pid)], 
                          capture_output=True)
        else:
            os.kill(pid, signal.SIGKILL)
    
    @staticmethod
    def find_processes(pattern):
        """查找匹配的进程"""
        if ProcessManager.is_windows():
            # Windows: 使用tasklist
            result = subprocess.run(
                ['tasklist', '/FI', f'IMAGENAME eq {pattern}'],
                capture_output=True,
                text=True
            )
            # 解析输出获取PID
            pids = []
            for line in result.stdout.split('\n'):
                if pattern.lower() in line.lower():
                    parts = line.split()
                    if len(parts) >= 2:
                        try:
                            pids.append(int(parts[1]))
                        except ValueError:
                            continue
            return pids
        else:
            # Linux: 使用pgrep
            result = subprocess.run(['pgrep', '-f', pattern],
                                  capture_output=True, text=True)
            return [int(pid) for pid in result.stdout.strip().split('\n') if pid]
    
    @staticmethod
    def start_background_process(command, log_file=None):
        """启动后台进程"""
        if ProcessManager.is_windows():
            # Windows后台进程
            if log_file:
                subprocess.Popen(
                    command,
                    stdout=open(log_file, 'w'),
                    stderr=subprocess.STDOUT,
                    creationflags=subprocess.CREATE_NEW_PROCESS_GROUP
                )
            else:
                subprocess.Popen(
                    command,
                    creationflags=subprocess.CREATE_NEW_PROCESS_GROUP
                )
        else:
            # Linux后台进程
            if log_file:
                subprocess.Popen(
                    command,
                    stdout=open(log_file, 'w'),
                    stderr=subprocess.STDOUT,
                    start_new_session=True
                )
            else:
                subprocess.Popen(command, start_new_session=True)
    
    @staticmethod
    def get_process_info(pid):
        """获取进程信息"""
        if ProcessManager.is_windows():
            try:
                result = subprocess.run(
                    ['tasklist', '/FI', f'PID eq {pid}', '/FO', 'CSV'],
                    capture_output=True,
                    text=True,
                    check=True
                )
                # 解析CSV格式输出
                lines = result.stdout.strip().split('\n')
                if len(lines) > 1:
                    parts = lines[1].split(',')
                    return {
                        'name': parts[0].strip('"'),
                        'pid': int(parts[1].strip('"')),
                        'memory': parts[4].strip('"')
                    }
            except (subprocess.CalledProcessError, ValueError, IndexError):
                pass
        else:
            try:
                result = subprocess.run(
                    ['ps', '-p', str(pid), '-o', 'comm,rss'],
                    capture_output=True,
                    text=True,
                    check=True
                )
                lines = result.stdout.strip().split('\n')
                if len(lines) > 1:
                    parts = lines[1].split()
                    return {
                        'name': parts[0],
                        'pid': pid,
                        'memory': f"{parts[1]} KB"
                    }
            except subprocess.CalledProcessError:
                pass
        return None

# tdx_connection.py
import sys
import os
import platform
import ctypes
from pathlib import Path
import logging

logger = logging.getLogger(__name__)

class TdxConnectionManager:
    """通达信连接管理器"""
    
    def __init__(self):
        self._dll = None
        self._initialized = False
        self.run_id = -1
        self._validate_environment()
    
    def _validate_environment(self):
        """验证运行环境"""
        # 1. 检查操作系统
        if platform.system() != 'Windows':
            raise RuntimeError("通达信API仅在Windows环境可用")
        
        # 2. 检查Python架构
        python_arch = platform.architecture()[0]  # '32bit' or '64bit'
        if python_arch != '64bit':
            logger.warning("推荐使用64位Python")
        
        # 3. 检查DLL文件存在
        dll_path = self._find_tdx_dll()
        if not dll_path or not dll_path.exists():
            raise RuntimeError(
                f"通达信DLL未找到: {dll_path}\n"
                "请确保通达信终端已正确安装"
            )
        
        logger.info(f"环境验证通过: {platform.system()} {python_arch}")
    
    def _find_tdx_dll(self):
        """查找通达信DLL路径"""
        # 优先级: 环境变量 > 配置文件 > 默认路径
        dll_path = os.environ.get('TDX_DLL_PATH')
        
        if dll_path:
            return Path(dll_path)
        
        # 常见通达信安装路径
        possible_paths = [
            Path(r"C:\新建文件夹\TdxW_HuaTai\TPythClient.dll"),
            Path(r"D:\新建文件夹\TdxW_HuaTai\TPythClient.dll"),
            Path(r"C:\通达信\TdxW_HuaTai\TPythClient.dll"),
            Path(r"k:\txdlianghua\TPythClient.dll"),  # 项目特定路径
        ]
        
        for path in possible_paths:
            if path.exists():
                logger.info(f"找到通达信DLL: {path}")
                return path
        
        return None
    
    def _find_tqcenter(self):
        """查找tqcenter.py路径"""
        # 优先级: 环境变量 > 配置文件 > 推算路径
        tqcenter_path = os.environ.get('TQ_CENTER_PATH')
        
        if not tqcenter_path:
            # 从DLL路径推算
            dll_path = self._find_tdx_dll()
            if dll_path:
                # DLL在TPythClient.dll，tqcenter.py在PYPlugins/user/
                tdx_root = dll_path.parent
                tqcenter_path = tdx_root / "PYPlugins" / "user"
        
        if tqcenter_path:
            tqcenter_file = Path(tqcenter_path) / "tqcenter.py"
            if tqcenter_file.exists():
                return str(tqcenter_file.parent)
        
        raise RuntimeError("无法找到tqcenter.py文件")
    
    def load_dll(self):
        """加载通达信DLL"""
        if self._dll is not None:
            return True
        
        try:
            dll_path = self._find_tdx_dll()
            
            # 设置DLL函数返回类型
            self._dll = ctypes.CDLL(str(dll_path))
            self._dll.InitConnect.restype = ctypes.c_char_p
            self._dll.GetTdxDataStr.restype = ctypes.c_char_p
            self._dll.TdxFuncMain.restype = ctypes.c_char_p
            
            logger.info("DLL加载成功")
            return True
            
        except Exception as e:
            logger.error(f"DLL加载失败: {e}")
            raise RuntimeError(f"通达信DLL加载失败: {e}")
    
    def initialize_tqcenter(self):
        """初始化tqcenter模块"""
        try:
            # 添加tqcenter路径到sys.path
            tqcenter_dir = self._find_tqcenter()
            if tqcenter_dir not in sys.path:
                sys.path.insert(0, tqcenter_dir)
            
            # 导入tqcenter
            from tqcenter import tq
            self.tq = tq
            
            # 初始化连接
            init_file = Path(tqcenter_dir) / "tqcenter.py"
            self.tq.initialize(str(init_file))
            
            self._initialized = True
            logger.info("tqcenter初始化成功")
            return True
            
        except ImportError as e:
            logger.error(f"tqcenter导入失败: {e}")
            raise RuntimeError(f"无法导入tqcenter: {e}")
        except Exception as e:
            logger.error(f"tqcenter初始化失败: {e}")
            raise RuntimeError(f"tqcenter初始化失败: {e}")
    
    def validate_connection(self):
        """验证通达信连接"""
        if not self._initialized:
            return False
        
        try:
            # 尝试获取简单数据验证连接
            test_data = self.tq.get_stock_list(market='1', list_type='0')
            return len(test_data) > 0
        except Exception as e:
            logger.warning(f"连接验证失败: {e}")
            return False
    
    def close(self):
        """关闭连接"""
        if self._initialized and hasattr(self, 'tq'):
            try:
                self.tq.close()
                logger.info("通达信连接已关闭")
            except Exception as e:
                logger.warning(f"关闭连接时出错: {e}")
            finally:
                self._initialized = False

# file_lock.py
import platform
import fcntl  # Linux
import msvcrt  # Windows
import os
from contextlib import contextmanager
import logging

logger = logging.getLogger(__name__)

class FileLock:
    """跨平台文件锁"""
    
    def __init__(self, file_path):
        self.file_path = file_path
        self.file_obj = None
        self.is_windows = platform.system() == 'Windows'
    
    def acquire(self):
        """获取文件锁"""
        try:
            self.file_obj = open(self.file_path, 'w')
            
            if self.is_windows:
                # Windows文件锁
                import pywintypes
                import win32file
                import win32con
                
                # 使用Windows API获取更可靠的锁
                handle = win32file.CreateFile(
                    self.file_path,
                    win32file.GENERIC_WRITE,
                    0,  # 不共享
                    None,
                    win32con.OPEN_ALWAYS,
                    0,
                    None
                )
                win32file.LockFileEx(
                    handle,
                    win32con.LOCKFILE_EXCLUSIVE_LOCK,
                    0xFFFF0000,
                    0xFFFFFFFF,
                    pywintypes.OVERLAPPED()
                )
                self.lock_handle = handle
                
            else:
                # Linux文件锁
                fcntl.flock(self.file_obj.fileno(), fcntl.LOCK_EX)
            
            logger.debug(f"文件锁获取成功: {self.file_path}")
            return True
            
        except Exception as e:
            logger.error(f"文件锁获取失败: {e}")
            if self.file_obj:
                self.file_obj.close()
            return False
    
    def release(self):
        """释放文件锁"""
        try:
            if self.is_windows and hasattr(self, 'lock_handle'):
                # Windows释放锁
                import win32file
                win32file.CloseHandle(self.lock_handle)
            elif self.file_obj:
                # Linux释放锁
                fcntl.flock(self.file_obj.fileno(), fcntl.LOCK_UN)
            
            if self.file_obj:
                self.file_obj.close()
            
            logger.debug(f"文件锁释放成功: {self.file_path}")
            
        except Exception as e:
            logger.warning(f"文件锁释放失败: {e}")
    
    def __enter__(self):
        return self.acquire()
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        self.release()

@contextmanager
def safe_file_write(file_path, content):
    """安全的文件写入上下文管理器"""
    lock = FileLock(file_path)
    try:
        with lock:
            with open(file_path, 'w', encoding='utf-8') as f:
                f.write(content)
            yield True
    except Exception as e:
        logger.error(f"文件写入失败: {e}")
        yield False

# db_pool.py
import duckdb
import threading
import time
from contextlib import contextmanager
import logging

logger = logging.getLogger(__name__)

class DatabaseConnectionPool:
    """DuckDB连接池"""
    
    def __init__(self, db_path, max_connections=5):
        self.db_path = db_path
        self.max_connections = max_connections
        self._connections = []
        self._lock = threading.Lock()
        self._in_use = set()
    
    @contextmanager
    def get_connection(self):
        """获取数据库连接"""
        conn = None
        try:
            # 等待可用连接
            while True:
                with self._lock:
                    if self._connections:
                        conn = self._connections.pop()
                        self._in_use.add(id(conn))
                        break
                    elif len(self._in_use) < self.max_connections:
                        conn = duckdb.connect(self.db_path)
                        self._in_use.add(id(conn))
                        break
                    else:
                        pass  # 等待其他连接释放
                
                if conn:
                    break
                
                time.sleep(0.1)  # 短暂等待
            
            yield conn
            
        finally:
            # 归还连接
            if conn:
                with self._lock:
                    self._in_use.discard(id(conn))
                    self._connections.append(conn)
    
    def close_all(self):
        """关闭所有连接"""
        with self._lock:
            for conn in self._connections:
                try:
                    conn.close()
                except:
                    pass
            self._connections.clear()
            self._in_use.clear()

# 全局连接池
_db_pool = None

def get_db_pool():
    """获取数据库连接池"""
    global _db_pool
    if _db_pool is None:
        _db_pool = DatabaseConnectionPool('data/duckdb/quant.db')
    return _db_pool

import logging
import logging.handlers
import os
import platform
from pathlib import Path
import win32evtlog  # Windows事件日志
import win32evtlogutil

class WindowsLogger:
    """Windows专用日志配置"""
    
    @staticmethod
    def setup_logger(name, log_file, level=logging.INFO):
        """配置Windows日志"""
        logger = logging.getLogger(name)
        logger.setLevel(level)
        
        # 文件处理器(Windows路径处理)
        log_path = Path(log_file)
        log_path.parent.mkdir(parents=True, exist_ok=True)
        
        # RotatingFileHandler
        file_handler = logging.handlers.RotatingFileHandler(
            log_file,
            maxBytes=10*1024*1024,  # 10MB
            backupCount=5,
            encoding='utf-8'
        )
        file_handler.setLevel(level)
        
        # 格式化器
        formatter = logging.Formatter(
            '%(asctime)s - %(name)s - %(levelname)s - %(message)s',
            datefmt='%Y-%m-%d %H:%M:%S'
        )
        file_handler.setFormatter(formatter)
        
        logger.addHandler(file_handler)
        
        # Windows事件日志(可选)
        if platform.system() == 'Windows':
            try:
                event_handler = win32evtlogutil.Logger(
                    'TdxQuant',
                    'TdxQuant量化交易系统'
                )
                logger.addHandler(event_handler)
            except:
                pass
        
        return logger

    @staticmethod
    def setup_console_logging(level=logging.INFO):
        """配置控制台日志(Windows编码处理)"""
        logger = logging.getLogger()
        logger.setLevel(level)
        
        # 控制台处理器
        console_handler = logging.StreamHandler()
        console_handler.setLevel(level)
        
        # 格式化器
        formatter = logging.Formatter(
            '%(asctime)s - %(levelname)s - %(message)s',
            datefmt='%H:%M:%S'
        )
        console_handler.setFormatter(formatter)
        
        logger.addHandler(console_handler)
        
        return logger

# windows_monitor.py
import psutil
import platform
import logging
from datetime import datetime
import json

logger = logging.getLogger(__name__)

class WindowsPerformanceMonitor:
    """Windows性能监控"""
    
    def __init__(self):
        self.is_windows = platform.system() == 'Windows'
    
    def get_system_info(self):
        """获取系统信息"""
        info = {
            'platform': platform.system(),
            'python_version': platform.python_version(),
            'architecture': platform.architecture(),
            'timestamp': datetime.now().isoformat()
        }
        
        if self.is_windows:
            # Windows特定信息
            import win32api
            import win32con
            
            info['windows_version'] = platform.version()
            info['computer_name'] = platform.node()
            
            # CPU信息
            info['cpu_count'] = psutil.cpu_count()
            info['cpu_freq'] = psutil.cpu_freq()._asdict()
            
            # 内存信息
            mem = psutil.virtual_memory()
            info['memory'] = {
                'total': mem.total,
                'available': mem.available,
                'percent': mem.percent
            }
            
            # 磁盘信息
            info['disk'] = {}
            for partition in psutil.disk_partitions():
                try:
                    usage = psutil.disk_usage(partition.mountpoint)
                    info['disk'][partition.mountpoint] = {
                        'total': usage.total,
                        'used': usage.used,
                        'free': usage.free,
                        'percent': usage.percent
                    }
                except:
                    pass
        
        return info
    
    def get_process_info(self, pid):
        """获取进程信息"""
        try:
            process = psutil.Process(pid)
            return {
                'pid': pid,
                'name': process.name(),
                'status': process.status(),
                'cpu_percent': process.cpu_percent(),
                'memory_info': process.memory_info()._asdict(),
                'num_threads': process.num_threads(),
                'create_time': datetime.fromtimestamp(process.create_time()).isoformat()
            }
        except psutil.NoSuchProcess:
            return None
    
    def check_healthy(self):
        """健康检查"""
        health = {
            'timestamp': datetime.now().isoformat(),
            'status': 'healthy'
        }
        
        # CPU检查
        cpu_percent = psutil.cpu_percent(interval=1)
        if cpu_percent > 90:
            health['status'] = 'warning'
            health['cpu_warning'] = f'CPU使用率过高: {cpu_percent}%'
        
        # 内存检查
        mem = psutil.virtual_memory()
        if mem.percent > 90:
            health['status'] = 'warning'
            health['memory_warning'] = f'内存使用率过高: {mem.percent}%'
        
        # 磁盘检查
        for partition in psutil.disk_partitions():
            try:
                usage = psutil.disk_usage(partition.mountpoint)
                if usage.percent > 90:
                    health['status'] = 'warning'
                    health['disk_warning'] = f'磁盘空间不足: {partition.mountpoint} {usage.percent}%'
            except:
                pass
        
        return health

# 定期监控

class WindowsCompatibilityTest(unittest.TestCase):

        cls.project_root = Path(__file__).parent.parent
    
    def test_path_handling(self):
