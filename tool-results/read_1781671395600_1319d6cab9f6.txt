   649→#### 4. 什么是量化交易（概念说明文档，无 API）
   650→
   651→5 步流程：投资想法 → 可执行策略 → 代码程序 → 回测/模拟验证 → 实盘交易并持续优化。
   652→
   653→策略三要素：**Security（品种）+ Condition（条件）+ Quantity（数量）**。
   654→
   655→文档内含完整金叉死叉策略回测代码示例（vectorbt）：
   656→```python
   657→import pandas as pd
   658→import vectorbt as vbt
   659→from tqcenter import tq
   660→
   661→tq.initialize(__file__)
   662→
   663→# 解决 pandas future warning
   664→pd.set_option('future.no_silent_downcasting', True)
   665→
   666→# ========================= 核心配置（用户可直接修改这里）=========================
   667→target_start = '20240930'  # 【目标回测开始时间】（真正想回测的起始日）
   668→target_end = '20250930'    # 【目标回测结束时间】
   669→stock_code_list = ['688318.SH']     # 股票代码
   670→window = 5         # MA指标周期（如MA5、MA10、MA20，改这里自动适配历史数据）
   671→# ================================================================================
   672→
   673→start_time = (pd.to_datetime(target_start) - pd.Timedelta(days=window + 10)).strftime('%Y%m%d')
   674→
   675→# 1.获取价格数据
   676→df_real = tq.get_market_data(
   677→    field_list=['Close', 'Open'],
   678→    stock_list=stock_code_list,
   679→    start_time=start_time,
   680→    end_time=target_end,
   681→    dividend_type='front',
   682→    period='1d',
   683→    fill_data=True
   684→)
   685→close_df = tq.price_df(df_real, 'Close', column_names=stock_code_list)
   686→open_df = tq.price_df(df_real, 'Open', column_names=stock_code_list)
   687→
   688→
   689→# 2.买卖信号计算与生成
   690→ma5_dynamic = vbt.MA.run(close_df, window=window).ma
   691→ma5_dynamic.columns = close_df.columns
   692→
   693→entries_raw = close_df.vbt.crossed_above(ma5_dynamic)
   694→exits_raw = close_df.vbt.crossed_below(ma5_dynamic)
   695→
   696→# 信号移位+1
   697→entries_df = entries_raw.shift(1).fillna(False).astype(bool)
   698→exits_df = exits_raw.shift(1).fillna(False).astype(bool)
   699→
   700→
   701→# 3. 执行回测
   702→portfolio = vbt.Portfolio.from_signals(
   703→    close=close_df,             # 净值计算用未复权收盘价
   704→    entries=entries_df,              # 延迟后的买入信号
   705→    exits=exits_df,                  # 延迟后的卖出信号
   706→    price=open_df,                # 含滑点的成交价格
   707→    init_cash=100000,            #  初始资金10万元
   708→    fees=0.0003,                  # 手续费0.03%（双边）
   709→    freq='D',                     # 日线频率
   710→    size_granularity=100          # A股最小交易单位100股
   711→)
   712→
   713→
   714→# 4. 输出回测结果
   715→print(f"\n======投资组合回测表现=====")
   716→print(portfolio.stats())
   717→print(f"\n======投资组合回测记录======")
   718→print(portfolio.trades.records_readable)
   719→```

   833→  print_to_tdx(df_list:       list[pd.DataFrame] = [],
   834→               sp_name:       str  = "",
   835→               xml_filename:  str  = "",

   837→               vertical:      int  = None,

   986→**操作步骤**：

   990→**策略说明**：如果运行时间点价格高出昨收 5%，则进入涨幅选股板块，否则清空该板块。
   991→
   992→**关键参数**：
   993→| 参数 | 默认值 | 说明 |
   994→|------|--------|------|
   995→| `start_time` | `"20251025"` | 数据起始日期 |
   996→| `target_gain` | `5.0` | 目标涨幅阈值（%），可按需修改 |
   997→| `target_block_name` | `'ZFXG'` | 自定义板块简称，需与第一步创建的板块一致 |
   998→| `dividend_type` | `'front'` | 前复权 |
   999→| `period` | `'1d'` | 日线周期 |

  1001→**完整代码（原样保留）**：
  1002→```python
  1003→# 策略说明：如果运行时间点价格高出昨收5%, 则进入涨幅选股板块，否则清空该板块
  1004→import pandas as pd
  1005→import numpy as np
  1006→from datetime import datetime
  1007→from tqcenter import tq
  1008→
  1009→# 初始化tq
  1010→tq.initialize(__file__)
  1011→

  1013→batch_codes = tq.get_stock_list_in_sector('通达信88')     # 目标板块
  1014→start_time = "20251025"                                  # 数据起始日期
  1015→target_end = datetime.now().strftime("%Y%m%d")           # 数据结束日期（当前日期）
  1016→target_gain = 5.0                                        # 目标涨幅（%），可修改
  1017→target_block_name = 'ZFXG'                               # 目标自定义板块简称
  1018→
  1019→# 2. 获取并整理收盘价数据
  1020→df_real = tq.get_market_data(
  1021→    field_list=['Close'],
  1022→    stock_list=batch_codes,
  1023→    start_time=start_time,
  1024→    end_time=target_end,
  1025→    dividend_type='front',  # 前复权
  1026→    period='1d',            # 日线
  1027→    fill_data=True          # 填充缺失数据
  1028→)

  1031→
  1032→# 3. 核心：计算当日相较于昨日的涨幅（%）
  1033→# 昨日收盘价（向下平移1行）
  1034→prev_close = close_df.shift(1)
  1035→# 计算涨幅：(当日收盘价 - 昨日收盘价) / 昨日收盘价 × 100%
  1036→daily_gain = (close_df - prev_close) / prev_close * 100
  1037→
  1038→# 4. 筛选符合条件的股票（最新交易日涨幅超target_gain%）
  1039→latest_date = daily_gain.index[-1]              # 最新交易日
  1040→latest_daily_gain = daily_gain.loc[latest_date] # 每只股票最新交易日的涨幅
  1041→# 筛选条件：涨幅 > target_gain%（排除NaN，避免数据异常）
  1042→target_stocks = latest_daily_gain[latest_daily_gain > target_gain].sort_values(ascending=False)
  1043→target_stocks_list = target_stocks.index.tolist()  # 提取符合条件的股票代码列表
  1044→
  1045→# 5. 结果输出与自定义板块操作（可按需注释）
  1046→print(f"\n=== 筛选结果（当日涨幅＞{target_gain}%）===")
  1047→if not target_stocks.empty:
  1048→    # ===================== 模块1：打印筛选结果 =====================
  1049→    print("【模块1：打印筛选结果】")
  1050→    print(f"符合条件的股票共 {len(target_stocks)} 只：")
  1051→    print(f"{'股票代码':<12} {'昨日收盘价':<12} {'当日收盘价':<12} {'当日涨幅':<10}")
  1052→    print("-" * 50)
  1053→    for stock_code, gain in target_stocks.items():
  1054→        prev_price = prev_close.loc[latest_date, stock_code]
  1055→        curr_price = close_df.loc[latest_date, stock_code]
  1056→        print(f"{stock_code:<12} {prev_price:<12.2f} {curr_price:<12.2f} {gain:<.2f}%")
  1057→    print("-" * 50)
  1058→
  1059→    # ===================== 模块2：添加至自定义板块 =====================
  1060→    try:
  1061→        print("【模块2：自定义板块操作】")
  1062→        tq.send_user_block(block_code=target_block_name, stocks=target_stocks_list, show=True)
  1063→        print(f"已成功将股票添加至自定义板块「{target_block_name}」")
  1064→    except Exception as e:
  1065→        print(f"添加自定义板块失败：{e}")
  1066→    print("-" * 50)
  1067→
  1068→else:
  1069→    # ===================== 模块1：打印空结果 =====================
  1070→    print("【模块1：打印筛选结果】")
  1071→    print(f"暂无当日涨幅＞{target_gain}%的股票")
  1072→    print("-" * 50)
  1073→
  1074→    # ===================== 模块2：清空自定义选板块 =====================
  1075→    try:
  1076→        print("【模块2：自定义板块操作】")
  1077→        tq.send_user_block(block_code=target_block_name, stocks=[], show=True)
  1078→        print(f"已清空自定义板块「{target_block_name}」")
  1079→    except Exception as e:
  1080→        print(f"清空自定义板块失败：{e}")
  1081→    print("-" * 50)
  1082→```

  1699→#### 板块管理完整能力（自定义板块 CRUD + 成份股管理）
  1700→
  1701→| 能力 | API | 说明 |
  1702→|------|-----|------|
  1703→| 列出自定义板块 | `tq.get_user_sector()` | 返回 `[{Code, Name}]` |

  1706→| 删除板块 | `tq.delete_sector(block_code)` | 删除整个板块 |
  1707→| 清空成份股 | `tq.clear_sector(block_code)` | 仅清股票，板块保留 |

  1711→**板块成份股读取三类接口对比**：
  1712→
  1713→| 场景 | API | 备注 |
  1714→|------|-----|------|
  1715→| 系统大类（全 A、沪深 300、可转债、ETF 等） | `tq.get_stock_list('xx')` | `list_type` 枚举 0~103，最强大 |
  1716→| 板块指数成份股（如钛金属 `880xxx.SH`） | `tq.get_stock_list_in_sector(code)` | 支持 code 或名称 |
  1717→| 自定义板块成份股 | `tq.get_stock_list_in_sector(code, block_type=1)` | 必须 `block_type=1` |
  1718→| 获取全部板块指数列表 | `tq.get_sector_list()` | 等价于 `get_stock_list('10')` |
  1719→
  1720→#### 可转债/ETF 字段
  1721→
  1722→- **可转债 `get_kzz_info`**：单只返回 dict，25 字段覆盖**转股价 / 转股价值 / 溢价率 / 强赎触发价 / 回售触发价 / 剩余规模 / 评级 / 关键日期 / 正股现价 / 转债现价**，是可转债监控全套字段。
  1723→- **ETF `get_trackzs_etf_info`**：按指数反查跟踪 ETF 列表，7 字段覆盖**现价 / 昨收 / IOPV净值 / 净额 / 规模**，用于指数套利、ETF 流动性筛选。
  1724→
  1725→#### 监控系统对接要点
  1726→1. **股票池回写闭环**：选股结果 → `clear_sector` → `send_user_block` → 通达信客户端即时显示，是程序选股与人工盯盘衔接的关键链路。
  1727→2. **`send_user_block` 是追加语义**：每日回写必须先 clear，否则股票堆积。
  1728→3. **`block_code` 必须预先存在**：`send_user_block` 不会自动创建板块，需先 `create_sector`。
  1729→4. **`ZXG`/`TJG` 是系统保留板块**：自选股 / 临时条件股可直接使用，无需 create。
  1730→5. **可转债监控**：通过 `get_stock_list('32')` 拿全市场可转债列表，再循环 `get_kzz_info` 拉单只详情，可构建全市场可转债监控表。
  1731→6. **ETF 选流动性**：`get_trackzs_etf_info` 返回多只 ETF，按 `Sz`（规模）和 `Zgb`（净额）排序可挑出最具流动性的标的。

  1951→#### 1. 获取快照数据（get_market_snapshot）
  1952→- 函数: `tq.get_market_snapshot(stock_code: str, field_list: List = []) -> Dict`
  1953→- 参数:
  1954→  - `stock_code` (str, 必填): 证券代码，如 `'688260.SH'`
  1955→  - `field_list` (List[str], 可选, 默认 `[]`): 字段筛选，传空返回全部
  1956→- 返回: `Dict`（单只股票的快照 dict，含 `ErrorId` 字段）。字段说明：
  1957→  - `ItemNum` (str): 快照笔数
  1958→  - `LastClose` (str): 前收盘价
  1959→  - `Open` / `Max` / `Min` / `Now` (str): 开盘/最高/最低/现价
  1960→  - `Volume` (str): 总手
  1961→  - `NowVol` (str): 现手
  1962→  - `Amount` (str): 总成交金额
  1963→  - `Inside` (str): 内盘（板块指数时为跌停家数）
  1964→  - `Outside` (str): 外盘（板块指数时为涨停家数）
  1965→  - `TickDiff` (str): 笔涨跌
  1966→  - `InOutFlag` (str): 内外盘标志：0 Buy / 1 Sell / 2 Unknown
  1967→  - `Jjjz` (str): 基金净值
  1968→  - `Buyp` / `Buyv` (List[str]): 五档买价 / 五档买盘量
  1969→  - `Sellp` / `Sellv` (List[str]): 五档卖价 / 五档卖盘量
  1970→  - `UpHome` / `DownHome` (str): 上涨/下跌家数（指数有效）
  1971→  - `Before5MinNow` (str): 5 分钟前价格
  1972→  - `Average` (str): 均价
  1973→  - `XsFlag` (str): 小数位数
  1974→  - `Zangsu` (str): 涨速
  1975→  - `ZAFPre3` (str): 3 日涨幅
  1976→  - `ErrorId` (str): 错误码（样本中为 `'0'`）
  1977→- 示例代码:
  1978→  ```python
  1979→  from tqcenter import tq
  1980→
  1981→  tq.initialize(__file__)
  1982→
  1983→  market_snapshot = tq.get_market_snapshot(stock_code='688260.SH', field_list=[])
  1984→  print(market_snapshot)
  1985→  ```
  1986→- 注意事项: 文档未单独列出注意事项；返回是单股 dict（非 DataFrame），五档买卖盘为 List[str]，需要时按 index 取值。

  2008→#### 3. 获取每天的股本数据（get_gb_info）
  2009→- 函数: `tq.get_gb_info(stock_code: str = '', date_list: List[str] = [], count: int = 1)`
  2010→- 参数:
  2011→  - `stock_code` (str, 必填): 股票代码
  2012→  - `date_list` (List[str], 必填): 日期数组（须从小到大排序）

  2126→#### 8. 获取 K 线行情（get_market_data）
  2127→- 函数: `tq.get_market_data(field_list: List[str] = [], stock_list: List[str] = [], period: str = '', start_time: str = '', end_time: str = '', count: int = -1, dividend_type: Optional[str] = None, fill_data: bool = True) -> Dict`
  2128→- 参数:
  2129→  - `field_list` (List[str], 可选, 默认 `[]`): 字段筛选，传空返回全部
  2130→  - `stock_list` (List[str], 必填): 证券代码列表
  2131→  - `period` (str, 必填): K 线周期（示例中用 `'1d'`，支持分钟/日/周/月等，需配合 `tqconst` 常量）
  2132→  - `start_time` (str, 可选): 起始时间（`'YYYYMMDD'`）
  2133→  - `end_time` (str, 可选): 结束时间（`'YYYYMMDD'`）
  2134→  - `count` (int, 可选, 默认 `-1`): 每只股票返回数据个数
  2135→  - `dividend_type` (Optional[str], 可选): `'none'`/`'front'`/`'back'`，不复权/前复权/后复权
  2136→  - `fill_data` (bool, 可选, 默认 `True`): 是否向后填充空缺数据
  2137→- 返回: `Dict[str, pd.DataFrame]`，key 为字段名（`Open`/`High`/`Low`/`Close`/...），value 为 `DataFrame`，index 为 `stock_list`、columns 为 `time_list`。各字段 DataFrame 维度与索引一致。字段说明：
  2138→  - `Date` (str): 日期
  2139→  - `Time` (str): 时间
  2140→  - `Open` / `High` / `Low` / `Close` (str): 开/高/低/收
  2141→  - `Volume` (str): 成交量
  2142→  - `Amount` (str): 成交额
  2143→  - `ForwardFactor` (str): 前复权因子（仅 `dividend_type='none'` 时有效）
  2144→  - `VolInStock` (str): 持仓量（仅期货返回有效值，非期货为 0）
  2145→- count 参数语义:
  2146→  - **count ≤ 0 或为空**:
  2147→    - 有 start 有 end: 取 start~end
  2148→    - 有 start 无 end: 从 start 取到最新
  2149→    - 无 start 有 end: 从最早取到 end
  2150→    - 都无: 取全部本地数据
  2151→  - **count > 0**:
  2152→    - 有 end: 从 end 往前取 n 根
  2153→    - 无 end: 从最新 K 线往前取 n 根
  2154→- 示例代码:
  2155→  ```python
  2156→  from tqcenter import tq
  2157→
  2158→  tq.initialize(__file__)
  2159→
  2160→  df = tq.get_market_data(
  2161→      field_list=[],
  2162→      stock_list=['688318.SH'],
  2163→      start_time='20251220',
  2164→      end_time='',
  2165→      count=1,
  2166→      dividend_type='none',
  2167→      period='1d',
  2168→      fill_data=True
  2169→  )
  2170→  print(df)
  2171→  ```
  2172→- 注意事项:
  2173→  - 仅 `dividend_type='none'` 时 `ForwardFactor` 返回有效值
  2174→  - 后复权只对返回的窗口内数据进行后复权
  2175→  - **一次最多返回 24000 条数据**，完整分钟线需分批多次调用
  2176→  - 若窗口内未发生权息变动，复权价 = 未复权价
  2177→  - 期货数据 `Amount=0`，非期货数据 `VolInStock=0`
  2178→  - 返回结构为「字段 → DataFrame」字典，而非传统 long-form DataFrame，多股票多周期时需手动 reshape

  2201→### 行情类 API 能力总结
  2202→
  2203→- **支持的行情类型**: 实时快照（`get_market_snapshot` 单股五档）、批量价量（`get_pricevol` 全市场前收/现价/成交量）、历史 K 线（`get_market_data`）、扩展行情/估值/资金流（`get_more_info`）、证券基本信息+财务摘要（`get_stock_info`）、板块归属（`get_relation`）、股本数据（`get_gb_info` 离散日 / `get_gb_info_by_date` 时间段）、新股/新债申购（`get_ipo_info`）。
  2204→- **K 线支持的周期**: 文档示例中明确出现 `period='1d'`（日线）。完整周期集合需参考 `tqcenter.tqconst` 常量（推测含 1m/5m/15m/30m/60m/1d/1w/1M 等，待 Task 通用函数模块确认）。
  2205→- **复权类型**: `none` / `front` / `back` 三种；仅 `none` 模式返回 `ForwardFactor` 前复权因子。
  2206→- **实时订阅能力**: 本批文档未出现订阅类（subscribe/push）API；当前均为主动查询（pull）模式，无 Tick 级推送。
  2207→- **历史数据深度**:
  2208→  - K 线单次最多返回 **24000 条**，分钟线需分批。
  2209→  - 股本数据可取「关键日」（`get_gb_info`）或「时间段全量」（`get_gb_info_by_date`，后者依赖日 K 线本地缓存）。
  2210→  - `get_more_info` 提供 52 周高低、近 2 年涨停/跌停/龙虎榜日期、年初至今涨幅等回溯指标。
  2211→- **批量能力**:
  2212→  - `get_pricevol` 支持传入完整 `stock_list`（配合 `get_stock_list(market='23')` 全市场扫描）。
  2213→  - `get_market_data` 的 `stock_list` 支持多股批量（返回结构 index=stock）。
  2214→  - 其余接口（`get_market_snapshot`/`get_more_info`/`get_stock_info`/`get_relation`）均为单股调用，需循环。
  2215→- **可替代哪些 V8 CSV**:
