    50→### 场景1: 执行选股策略并加入客户端自定义板块
    51→
    52→- **业务流程**:
    53→  1. 从板块 `通达信88` 获取股票池
    54→  2. 拉取历史日线 Close（前复权、填充缺失）
    55→  3. 用 pandas + numpy 计算每只股票"连续上涨天数"（核心技巧：`up_mask` 上涨日置 1、非上涨置 NaN → `ffill` → `cumsum` → 在非上涨日处重置计数）
    56→  4. 筛选"最新交易日连续上涨 ≥ N 天"的股票
    57→  5. 调用 `tq.create_sector` 创建自定义板块
    58→  6. 调用 `tq.send_user_block` 把股票列表推送到客户端板块（无符合则推空列表清空）
    59→  7. 调用 `tq.send_message` 把 MSG 提示推送到 TQ 策略管理器
    60→  8. 在客户端 `LZXG` 板块中查看结果
    61→
    62→- **完整代码示例**:
    63→  ```python
    64→  import pandas as pd
    65→  import numpy as np
    66→  from datetime import datetime
    67→  from tqcenter import tq
    68→
    69→  # 初始化tq
    70→  tq.initialize(__file__)
    71→
    72→  # 1. 基础配置（可修改项）
    73→  batch_codes = tq.get_stock_list_in_sector('通达信88')    # 目标板块
    74→  start_time = "20251025"                                  # 数据起始日期
    75→  target_end = datetime.now().strftime("%Y%m%d")           # 数据结束日期（当前日期）
    76→  N = 3                                                    # 目标连续上涨天数
    77→  block_code = 'LZXG'                                      # 自定义板块简称（必选）
    78→  block_name = '连涨选股'                                   # 自定义板块名称（必选）
    79→
    80→  # 2. 获取并整理收盘价数据
    81→  df_real = tq.get_market_data(
    82→      field_list=['Close'],
    83→      stock_list=batch_codes,
    84→      start_time=start_time,
    85→      end_time=target_end,
    86→      dividend_type='front',  # 前复权
    87→      period='1d',            # 日线
    88→      fill_data=True          # 填充缺失数据
    89→  )
    90→  # 转换为「日期×股票代码」的收盘价宽表
    91→  close_df = tq.price_df(df_real, 'Close', column_names=batch_codes)
    92→
    93→  # 3. 标记每日是否上涨（核心判断逻辑）
    94→  is_up = close_df > close_df.shift(1)  # True=当日上涨，False=当日非上涨
    95→
    96→  # 4. 核心：计算连续上涨天数
    97→  # 步骤1：上涨日标记为1，非上涨日标记为NaN
    98→  up_mask = np.where(is_up, 1, np.nan)
    99→  up_mask_df = pd.DataFrame(up_mask, index=close_df.index, columns=close_df.columns)
   100→
   101→  # 步骤2：前向填充 → 连续上涨阶段的非上涨日（NaN）会被1填充
   102→  filled_df = up_mask_df.ffill()
   103→
   104→  # 步骤3：累计非NaN值的数量（初步计数）
   105→  consec_up_days = filled_df.notna().cumsum()
   106→
   107→  # 步骤4：非上涨日重置计数（关键步骤，实现"连续"效果）
   108→  reset_counts = consec_up_days.where(~is_up).ffill().fillna(0)
   109→  consec_up_days = (consec_up_days - reset_counts).astype(int)
   110→
   111→  # 5. 筛选符合条件的股票（连续上涨≥N天）
   112→  latest_date = consec_up_days.index[-1]  # 最新交易日
   113→  latest_consec_up = consec_up_days.loc[latest_date]  # 每只股票最新的连续上涨天数
   114→  target_stocks = latest_consec_up[latest_consec_up >= N].sort_values(ascending=False)
   115→  target_stocks_list = target_stocks.index.tolist()  # 提取符合条件的股票代码列表
   116→
   117→  # 6. 先创建自定义板块，再执行添加/清空操作
   118→  print(f"\n=== 筛选结果（连续上涨≥{N}天）===")
   119→  # 第一步：创建自定义板块
   120→  try:
   121→      tq.create_sector(block_code=block_code, block_name=block_name)
   122→      print(f"✅ 已成功创建自定义板块「{block_name}（{block_code}）」")
   123→  except Exception as e:
   124→      # 板块已存在时可能报错，此处捕获异常不中断流程
   125→      print(f"ℹ️  自定义板块创建提示：{e}（若提示已存在，可忽略此信息）")
   126→
   127→  # 第二步：处理板块成份股（添加/清空）
   128→  if not target_stocks.empty:
   129→      # 打印筛选结果
   130→      print(f"符合条件的股票共 {len(target_stocks)} 只：")
   131→      for stock_code, days in target_stocks.items():
   132→          print(f"{stock_code}：连续上涨 {days} 天")
   133→      
   134→      # 发送至自定义板块
   135→      try:
   136→          tq.send_user_block(block_code=block_code, stocks=target_stocks_list)
   137→          print(f"\n✅ 已成功将股票添加至自定义板块「{block_name}（{block_code}）」")
   138→      except Exception as e:
   139→          print(f"\n❌ 添加自定义板块失败：{e}")
   140→      
   141→      # 发送提示消息至TQ策略管理器
   142→      msg = f"MSG,筛选结果：{start_time}至{target_end}，连续上涨≥{N}天的股票共{len(target_stocks)}只，已添加至「{block_name}（{block_code}）」"
   143→      try:
   144→          tq.send_message(msg)
   145→          print("✅ 提示消息发送成功")
   146→      except Exception as e:
   147→          print(f"❌ 消息发送失败：{e}")
   148→  else:
   149→      # 无符合条件股票时清空板块
   150→      print("暂无符合条件的股票")
   151→      try:
   152→          tq.send_user_block(block_code=block_code, stocks=[])
   153→          print(f"✅ 已清空自定义板块「{block_name}（{block_code}）」")
   154→      except Exception as e:
   155→          print(f"❌ 清空自定义板块失败：{e}")
   156→      
   157→      # 发送空结果提示
   158→      msg = f"MSG,筛选结果：{start_time}至{target_end}，连续上涨≥{N}天的股票共0只，已清空「{block_name}（{block_code}）」"
   159→      try:
   160→          tq.send_message(msg)
   161→      except Exception as e:
   162→          print(f"❌ 消息发送失败：{e}")
   163→  ```

   173→#### 2.1 利用 MACD 公式筛选金叉信号（批量处理版）
   174→
   175→- **业务流程**: 全市场 → `formula_process_mul_zb` 一次算全市场 MACD → 取 DIF/DEA 最后 2 个值 → 判断"昨日 DIF<DEA，今日 DIF>=DEA" → 推送到客户端自选股
   176→- **完整代码**:
   177→  ```python
   178→  from tqcenter import tq
   179→
   180→  '''
   181→      利用此示例需要先在客户端下载全A股盘后数据，不然结果不准确
   182→      通过MACD指标公式选出最新交易日金叉的股票
   183→  '''
   184→
   185→  tq.initialize(__file__)
   186→
   187→  # 先获取A股全部股票
   188→  all_stocks = tq.get_stock_list(market='5')
   189→
   190→  print("正在处理，请等待...")
   191→  import time
   192→
   193→  # 开始计时
   194→  start_time = time.time()
   195→
   196→  macd_stocks = []
   197→  mul_zb_result = tq.formula_process_mul_zb(
   198→      formula_name='MACD',
   199→      formula_arg='12,26,9',
   200→      xsflag=6,
   201→      return_count=2,
   202→      return_date=False,
   203→      stock_list=all_stocks,
   204→      stock_period='1d',
   205→      count=100,
   206→      dividend_type=1)
   207→
   208→  if mul_zb_result:
   209→      for key in mul_zb_result:
   210→          if key != "ErrorId":
   211→              if len(mul_zb_result[key]['DIF']) >= 2 and len(mul_zb_result[key]['DEA']) >= 2:
   212→                  if float(mul_zb_result[key]['DIF'][-2]) < float(mul_zb_result[key]['DEA'][-2]) and float(mul_zb_result[key]['DIF'][-1]) >= float(mul_zb_result[key]['DEA'][-1]):
   213→                      macd_stocks.append(key)
   214→
   215→  print("今日MACD金叉股票列表：")
   216→  print(macd_stocks)
   217→  print("符合MACD金叉条件的股票数量：", len(macd_stocks))
   218→  # 结束计时
   219→  end_time = time.time()
   220→
   221→  # 计算时间差
   222→  execution_time = end_time - start_time
   223→  print(f"执行时间: {execution_time:.6f} 秒")
   224→  print(f"执行时间: {execution_time * 1000:.2f} 毫秒")
   225→
   226→  zxg_result = tq.send_user_block(block_code='', stocks=macd_stocks)
   227→  ```

   230→#### 2.2 利用 MACD 公式筛选金叉信号（for 循环版）
   231→
   232→- **业务流程**: 逐只股票 → `formula_set_data_info` 设置股票数据上下文 → `formula_zb` 取 MACD → 判断金叉。适合小批量调试，可对比性能。
   233→- **完整代码**:
   234→  ```python
   235→  from tqcenter import tq
   236→
   237→  '''
   238→      利用此示例需要先在客户端下载全A股盘后数据，不然结果不准确
   239→      通过MACD指标公式选出最新交易日金叉的股票
   240→  '''
   241→  tq.initialize(__file__)
   242→
   243→  # 先获取A股全部股票
   244→  all_stocks = tq.get_stock_list(market='5')
   245→
   246→  print("正在处理，请等待...")
   247→
   248→  import time
   249→
   250→  # 开始计时
   251→  start_time = time.time()
   252→
   253→  macd_stocks = []
   254→
   255→  for stock in all_stocks:
   256→      try:
   257→          # 1. 设置股票数据
   258→          tq.formula_set_data_info(
   259→              stock_code=stock,
   260→              stock_period='1d',
   261→              count=100,  # 需要足够的数据计算MACD
   262→              dividend_type=1  # 前复权
   263→          )
   264→
   265→          # 2. 获取MACD指标
   266→          macd_result = tq.formula_zb(
   267→              formula_name='MACD',
   268→              formula_arg='12,26,9',
   269→              xsflag=6
   270→          )
   271→
   272→          # 3. 获取DIF和DEA值，判断金叉
   273→          if macd_result and 'Data' in macd_result:
   274→              dif_values = macd_result['Data']['DIF']
   275→              dea_values = macd_result['Data']['DEA']
   276→
   277→              if len(dif_values) >= 2 and len(dea_values) >= 2:
   278→                  dif_prev = float(dif_values[-2])  # 前一天的DIF
   279→                  dif_now = float(dif_values[-1])   # 今天的DIF
   280→                  dea_prev = float(dea_values[-2])  # 前一天的DEA
   281→                  dea_now = float(dea_values[-1])   # 今天的DEA
   282→
   283→                  # MACD金叉信号：昨天DIF<DEA，今天DIF>=DEA
   284→                  if dif_prev < dea_prev and dif_now >= dea_now:
   285→                      macd_stocks.append(stock)
   286→                      print(f"MACD金叉信号: {stock}, DIF: {dif_prev:.4f}→{dif_now:.4f}, DEA: {dea_prev:.4f}→{dea_now:.4f}")
   287→
   288→      except Exception as e:
   289→          print(f"处理股票 {stock} 时出错: {e}")
   290→          continue
   291→
   292→  print("今日MACD金叉股票列表：")
   293→  print(macd_stocks)
   294→  print("符合MACD金叉条件的股票数量：", len(macd_stocks))
   295→
   296→  # 结束计时
   297→  end_time = time.time()
   298→
   299→  # 计算时间差
   300→  execution_time = end_time - start_time
   301→
   302→  print(f"执行时间: {execution_time:.6f} 秒")
   303→  print(f"执行时间: {execution_time * 1000:.2f} 毫秒")
   304→
   305→  zxg_result = tq.send_user_block(block_code='', stocks=macd_stocks)
   306→  ```
   307→- **涉及 API**: `tq.get_stock_list` / `tq.formula_set_data_info` / `tq.formula_zb` / `tq.send_user_block`

   713→### 场景3: 订阅行情涨幅突破实时预警（实时监控关键参考）
   714→
   715→- **订阅模式**:
   716→  - 从板块（`通达信88`）取股票池，**分批 50 只一组**调 `tq.subscribe_hq`（规避单次订阅过多/订阅上限）
   717→  - 回调函数 `price_rise_callback` 接收 JSON 字符串（含 `Code`/`ErrorId`）
   718→  - 回调内调 `tq.get_full_tick(code)` 取最新快照（`Now`/`LastClose`/`Volume`）
   719→  - 计算 `rise_rate = (latest_price - pre_close) / pre_close * 100`
   720→  - 超过 `PRICE_RISE_THRESHOLD=5.0%` 触发预警
   721→- **回调机制**:
   722→  - **前置过滤**: 非监控股票、已触发股票、`ErrorId != "0"`、无效价格 → 直接 return
   723→  - **防抖**: `last_warn_time` defaultdict 记录每只股票上次触发时间，10 秒内不重复推送
   724→  - **首次触发即退订**: 把股票加入 `TRIGGERED_STOCKS` set → 调 `unsubscribe_single_stock` 取消该股票订阅（避免重复监控/推送）
   725→  - **信号处理**: `signal.signal(SIGINT, signal_handler)` 捕获 Ctrl+C → 强制 `unsubscribe_stocks` 清理资源 → `sys.exit(0)`
   726→- **完整代码**:

   798→      """

   812→                  total_success = False

   814→                  print(f"[{datetime.now().strftime('%H:%M:%S')}] 第{i//batch_size + 1}批订阅成功：{sub_res}")

   834→          return False
   835→      except Exception as e:

   839→
   840→  # ===================== 核心回调函数 =====================
   841→  def price_rise_callback(data_str):
   842→      try:
   843→          code_json = json.loads(data_str)
   844→          code = code_json.get('Code')
   845→
   846→          # 前置过滤：无效数据/非监控股票/已触发过的股票
   847→          if (code_json.get('ErrorId') != "0" or not code) or \
   848→             (code not in SUBSCRIBE_CODES) or \
   849→             (code in TRIGGERED_STOCKS):
   850→              return
   851→
   852→          # 获取最新行情数据
   853→          report_ptr = tq.get_full_tick(code)
   854→
   855→          latest_price = 0.0
   856→          pre_close = 0.0
   857→
   858→          if report_ptr:
   859→              latest_price = round(float(report_ptr['Now']), 2)
   860→              pre_close = round(float(report_ptr['LastClose']), 2)
   861→
   862→              if pre_close <= 0 and latest_price > 0:
   863→                  pre_close = latest_price - 0.01
   864→
   865→          # 过滤最新价/昨收价无效的情况
   866→          if latest_price <= 0 or pre_close <= 0:
   867→              return
   868→
   869→          # 计算涨幅
   870→          rise_rate = round(((latest_price - pre_close) / pre_close) * 100, 2) if pre_close > 0 else 0.0
   871→
   872→          # 仅处理满足涨幅阈值+防抖的情况
   873→          if rise_rate > PRICE_RISE_THRESHOLD:
   874→              current_time = int(time.time())
   875→              if current_time - last_warn_time[code] < ANTI_SHAKE_SECONDS:
   876→                  return
   877→
   878→              # 标记为已触发，后续不再处理
   879→              TRIGGERED_STOCKS.add(code)
   880→              last_warn_time[code] = current_time
   881→
   882→              # 取消该股票的订阅（不再监控）
   883→              unsubscribe_single_stock(code)
   884→
   885→              # 发送预警
   886→              warn_time = datetime.now().strftime("%Y%m%d%H%M%S")
   887→              reason = "涨幅突破"
   888→
   889→              try:
   890→                  # 成交量用实际值，无则填0
   891→                  volume = report_ptr.get('Volume', '0') if report_ptr else '0'
   892→                  warn_res = tq.send_warn(
   893→                      stock_list=[code],
   894→                      time_list=[warn_time],
   895→                      price_list=[str(latest_price)],
   896→                      close_list=[str(pre_close)],
   897→                      volum_list=[volume])
   898→                  print(f"[{datetime.now().strftime('%H:%M:%S')}] 预警已发送：{code} 涨幅{rise_rate}%")
   899→              except Exception as e:
   900→                  print(f"[{datetime.now().strftime('%H:%M:%S')}] 发送预警失败：{e}")
   901→
   902→      except Exception as e:
   903→          print(f"[{datetime.now().strftime('%H:%M:%S')}] 回调处理异常：{e}")
   904→
   905→
   906→  # ===================== 主函数 =====================
   907→  def main():
   908→      # 注册信号处理器
   909→      signal.signal(signal.SIGINT, signal_handler)
   910→
   911→      # 初始化通达信
   912→      tq.initialize(__file__)
   913→
   914→      # 获取监控股票列表
   915→      global SUBSCRIBE_CODES
   916→      SUBSCRIBE_CODES = get_valid_stock_codes(SECTOR_NAMES)
   917→
   918→      if not SUBSCRIBE_CODES:
   919→          print("未获取到有效股票代码，程序退出")
   920→          return
   921→
   922→      # 分批订阅
   923→      print(f"\n[{datetime.now().strftime('%H:%M:%S')}] 开始订阅{len(SUBSCRIBE_CODES)}只股票...")
   924→      batch_subscribe(SUBSCRIBE_CODES, BATCH_SUBSCRIBE_SIZE)
   925→
   926→      # 保持程序运行
   927→      print(f"\n[{datetime.now().strftime('%H:%M:%S')}] 订阅完成，进入监控模式（按 Ctrl+C 退出）...")
   928→      while not EXIT_FLAG:
   929→          time.sleep(1)
   930→
   931→      print("程序已退出")
   932→
   933→
   934→  if __name__ == "__main__":
   935→      main()
   936→  ```

   949→### 场景4: 计算调仓信号并快速买卖（实盘交易关键参考）
   950→
   951→> ⚠️ 注意：本文档名为"快速买卖"但实际只演示了**信号计算 + MSG 推送 + send_warn 预警**，**没有出现真正下单 API**（如 `place_order`）。真正的下单是通过 `send_warn` 推送买卖预警到客户端，用户**双击预警记录打开闪电下单**完成交易。这是 TdxQuant 当前的"半自动"模式。
   952→
   953→- **下单 API**: 实际无直接下单 API；通过 `send_warn(bs_flag_list=['0'买/'1'卖])` 推送买卖预警，用户在客户端双击预警调用闪电下单
   954→- **委托管理**: 文档未涉及委托查询/撤单 API（推测需查 d客户端操作类 或 通用函数 文档）
   955→- **业务流程**:
   956→  1. 取板块 `通达信88` 股票池，回算 start_date = 今天 - (2*N+20) 天
   957→  2. `get_market_data` 取前复权 Close → `price_df` 转宽表
   958→  3. 用 `vbt.MA.run` 算 N 日均线；`close_df.vbt.crossed_above(ma)` / `crossed_below(ma)` 生成买入/卖出信号
   959→  4. 遍历股票，提取 latest_date 当天的 buy_signals / sell_signals（含 today_close / prev_close / ma_price）
   960→  5. `send_message("MSG,...")` 把统计行+信号明细推送到 TQ 策略管理器
   961→  6. `send_warn(bs_flag_list='0'买/'1'卖, warn_type_list='1')` 推买卖预警到客户端
   962→- **完整代码**:
   963→  ```python
   964→  from datetime import datetime, timedelta
   965→  from tqcenter import tq as tdxdata
   966→  import vectorbt as vbt
   967→  import pandas as pd
   968→
   969→  # 初始化
   970→  tdxdata.initialize(__file__)
   971→  run_time = datetime.now()
   972→  run_time_str = run_time.strftime("%Y-%m-%d %H:%M:%S")
   973→  # 预警时间戳（格式：YYYYMMDDHHMMSS）
   974→  warn_time = run_time.strftime("%Y%m%d%H%M%S")
   975→
   976→  # ===================== 1. 配置参数 =====================
   977→  N = 5  # 均线周期
   978→  batch_codes = tdxdata.get_stock_list_in_sector('通达信88')
   979→  end_date = run_time.strftime("%Y%m%d")
   980→  start_date = (run_time - timedelta(days=2 * N + 20)).strftime("%Y%m%d")
   981→
   982→  # ===================== 2. 获取并处理数据 =====================
   983→  # 获取日线Close数据（保留完整索引用于日期筛选）
   984→  df_real = tdxdata.get_market_data(
   985→      field_list=['Close'],
   986→      stock_list=batch_codes,
   987→      start_time=start_date,
   988→      end_time=end_date,
   989→      dividend_type='front',
   990→      period='1d',
   991→      fill_data=True
   992→  )
   993→  close_df = tdxdata.price_df(df_real, 'Close', column_names=batch_codes)
   994→
   995→  # 计算均线+生成信号
   996→  ma = vbt.MA.run(close_df, window=N).ma
   997→  ma.columns = close_df.columns
   998→  entries = close_df.vbt.crossed_above(ma)  # 上穿（买入）
   999→  exits = close_df.vbt.crossed_below(ma)    # 下穿（卖出）
  1000→  latest_date = close_df.index[-1]  # 今日日期（DataFrame最后一行）
  1001→  # 获取上一个工作日日期
  1002→  prev_date = close_df.index[-2] if len(close_df.index) >= 2 else latest_date
  1003→
  1004→  # ===================== 3. 筛选最新买卖信号 =====================
  1005→  buy_signals = {}
  1006→  sell_signals = {}
  1007→
  1008→  # 遍历股票筛选信号
  1009→  for code in batch_codes:
  1010→      # 确保股票有足够的交易数据
  1011→      if code not in close_df.columns:
  1012→          continue
  1013→      
  1014→      # 今日收盘价
  1015→      today_close = close_df.loc[latest_date, code]
  1016→      # 上一个工作日收盘价
  1017→      prev_close = close_df.loc[prev_date, code] if len(close_df.index) >= 2 else today_close
  1018→      
  1019→      # 买入信号：最新日期Close上穿均线
  1020→      if entries.loc[latest_date, code]:
  1021→          buy_signals[code] = {
  1022→              'today_close': round(today_close, 2),    # 今日close
  1023→              'prev_close': round(prev_close, 2),      # 上一个工作日close
  1024→              'ma_price': round(ma.loc[latest_date, code], 2)
  1025→          }
  1026→      # 卖出信号：最新日期Close下穿均线
  1027→      if exits.loc[latest_date, code]:
  1028→          sell_signals[code] = {
  1029→              'today_close': round(today_close, 2),    # 今日close
  1030→              'prev_close': round(prev_close, 2),      # 上一个工作日close
  1031→              'ma_price': round(ma.loc[latest_date, code], 2)
  1032→          }
  1033→
  1034→  # ===================== 4. 生成并发送MSG =====================
  1035→  def send_msg(content):
  1036→      msg = f"MSG,{content}"
  1037→      print(msg)
  1038→      try:
  1039→          tdxdata.send_message(msg)
  1040→      except Exception as e:
  1041→          print(f"发送失败: {e}")
  1042→
  1043→  # 统计行
  1044→  stat_line = (
  1045→      f"运行时间：{run_time_str}，均线周期：{N}天，"
  1046→      f"买入信号数：{len(buy_signals)} 只，卖出信号数：{len(sell_signals)} 只"
  1047→  )
  1048→
  1049→  print("\n=== MSG格式（TQ策略管理器显示区域）===")
  1050→  send_msg(stat_line)
  1051→
  1052→  # 处理买入信号
  1053→  if buy_signals:
  1054→      send_msg(f"=== 买入信号（Close上穿{N}日均线）===")
  1055→      for idx, (code, info) in enumerate(buy_signals.items(), 1):
  1056→          line = f"{idx}. {code}：买入信号，今日Close:{info['today_close']}，昨日Close:{info['prev_close']}"
  1057→          send_msg(line)
  1058→
  1059→  # 处理卖出信号
  1060→  if sell_signals:
  1061→      send_msg(f"=== 卖出信号（Close下穿{N}日均线）===")
  1062→      for idx, (code, info) in enumerate(sell_signals.items(), 1):
  1063→          line = f"{idx}. {code}：卖出信号，今日Close:{info['today_close']}，昨日Close:{info['prev_close']}"
  1064→          send_msg(line)
  1065→
  1066→  # 无信号的情况
  1067→  if not buy_signals and not sell_signals:
  1068→      send_msg(f"运行时间：{run_time_str}，均线周期：{N}天，无买入或卖出信号")
  1069→
  1070→  # ===================== 5. 调用send_warn接口发送预警 =====================
  1071→  def send_trade_warn():
  1072→      """发送买卖信号对应的预警（精简版，仅保留核心逻辑）"""
  1073→      # 合并所有信号用于发送预警
  1074→      all_signals = []
  1075→      if buy_signals:
  1076→          all_signals.extend([(code, info, '买入') for code, info in buy_signals.items()])
  1077→      if sell_signals:
  1078→          all_signals.extend([(code, info, '卖出') for code, info in sell_signals.items()])
  1079→      
  1080→      if not all_signals:
  1081→          print("\n无预警信息需要发送")
  1082→          return
  1083→      
  1084→      # 构造预警参数列表
  1085→      codes = []
  1086→      time_list = []
  1087→      price_list = []       # 今日close
  1088→      close_list = []       # 上一个工作日close
  1089→      volum_list = []
  1090→      bs_flag_list = []
  1091→      warn_type_list = []
  1092→      reason_list = []
  1093→      
  1094→      for code, info, trade_type in all_signals:
  1095→          codes.append(code)
  1096→          time_list.append(warn_time)
  1097→          price_list.append(str(info['today_close']))    # 替换为今日close
  1098→          close_list.append(str(info['prev_close']))     # 替换为上一个工作日close
  1099→          volum_list.append('0')
  1100→          bs_flag_list.append('0' if trade_type == '买入' else '1')
  1101→          warn_type_list.append('1')
  1102→          reason_list.append(f"{trade_type}信号")
  1103→      
  1104→      # 调用预警接口
  1105→      try:
  1106→          warn_res = tdxdata.send_warn(
  1107→              stock_list=codes,
  1108→              time_list=time_list,
  1109→              price_list=price_list,
  1110→              close_list=close_list,
  1111→              volum_list=volum_list,
  1112→              bs_flag_list=bs_flag_list,
  1113→              warn_type_list=warn_type_list,
  1114→              reason_list=reason_list,
  1115→              count=len(codes)
  1116→          )
  1117→          print(f"\n预警发送完成，共发送 {len(codes)} 条预警，接口返回：{warn_res}")
  1118→      except Exception as e:
  1119→          print(f"\n预警发送失败：{e}")
  1120→
  1121→  # 执行预警发送
  1122→  send_trade_warn()
  1123→
  1124→  print("\n所有消息发送完成！")
  1125→  tdxdata.close()
  1126→  ```
  1127→- **涉及 API**: `tq.initialize` / `tq.get_stock_list_in_sector` / `tq.get_market_data` / `tq.price_df` / `tq.send_message` / `tq.send_warn`（bs_flag: 0买1卖2未知；warn_type: 1表示买卖信号） / `tq.close`

  1131→### 场景5: VBT 简单回测（回测引擎参考）
  1132→
  1133→- **回测流程**:
  1134→  1. 多取 `window + 10` 天历史数据用于 MA 预热
  1135→  2. `tq.get_market_data(field_list=['Close','Open'], dividend_type='front')` → `price_df` 转宽表
  1136→  3. `vbt.MA.run(close_df, window=window).ma.ffill()` 计算均线
  1137→  4. **信号延迟一天** `close_df.vbt.crossed_above(ma5).shift(1).fillna(False).astype(bool)` —— 防止未来函数
  1138→  5. `vbt.Portfolio.from_signals(close=close_df, entries=, exits=, price=open_df, init_cash=100000, fees=0.0003, freq='D', size_granularity=100)` 执行回测（A 股最小 100 股）
  1139→  6. `portfolio[stock].plot().show()` 交互式图形
  1140→  7. `portfolio.stats()` 输出统计指标；`portfolio.trades.records_readable` 输出交易记录
  1141→- **注意**: VBT 的 `Portfolio.from_signals` **不支持分红送股等权益变动**，官方明确说仅做示例。如需精确回测，应用前复权价格且仅做短线回测，或改用场景6 的 `send_bt_data` 自定义回测。

  1172→  )

  1175→

  1177→  ma5_dynamic = vbt.MA.run(close_df, window=window).ma.ffill()

  1191→      exits=exits_df,             # 延迟后的卖出信号

  1388→  def calculate_ema(series, window):
  1389→      """计算指数移动平均"""
  1390→      return series.ewm(span=window, adjust=False).mean()
  1391→
  1392→  def calculate_sma(series, window):
  1393→      """计算简单移动平均"""
  1394→      return series.rolling(window=window).mean()
  1395→
  1396→  def calculate_llv(series, window):
  1397→      """计算周期内最低值"""
  1398→      return series.rolling(window=window).min()
  1399→
  1400→  def calculate_hhv(series, window):

  1402→      return series.rolling(window=window).max()
  1403→
  1404→  def calculate_ma(series, window):

  1407→
  1408→  def ref(series, periods):
  1409→      """引用若干周期前的数据"""

  1412→  def calculate_cross_signal(fast_series, slow_series):
  1413→      """
  1414→      计算金叉信号序列。
  1415→      规则：当快线从下方上穿慢线时，标记为1（金叉），否则为0。

  1417→      """
  1418→      # 判断条件：今日快线 > 慢线，且昨日快线 <= 慢线
  1419→      cross_up = (fast_series > slow_series) & (ref(fast_series, 1) <= ref(slow_series, 1))
  1420→      return cross_up.astype(int)

  1422→  def get_benchmark_data(benchmark_code='000300.SH', count=60):

  1424→      获取基准品种数据

  1452→              benchmark_annual_return = 0

  1473→  def calculate_daily_statistics(df, benchmark_data, initial_capital=100000, fee_rate=0.0003):
  1474→      """
  1475→      计算每日的回测统计指标序列
  1476→      输入：
  1477→          df: 包含价格数据、买卖信号的DataFrame
  1478→          benchmark_data: 基准数据字典
  1479→          initial_capital: 初始资金
  1480→          fee_rate: 手续费率（双边）
  1481→      输出：包含每日统计指标的DataFrame
  1482→      """
  1483→      # 初始化变量
  1484→      capital = initial_capital
  1485→      position = 0  # 持仓数量
  1486→      rest_cash = capital  # 剩余现金
  1487→      hold = False  # 是否持仓
  1488→      
  1489→      # 创建结果列表
  1490→      daily_stats = []
  1491→      open_count = 0  # 累计开仓次数
  1492→      close_count = 0  # 累计平仓次数
  1493→      win_count = 0  # 累计盈利交易次数
  1494→      trade_records = []  # 交易记录
  1495→      
  1496→      # 获取基准数据
  1497→      benchmark_close = benchmark_data['close']
  1498→      benchmark_net_value = benchmark_data['net_value']
  1499→      
  1500→      # 确保基准数据与策略数据时间对齐
  1501→      if len(benchmark_close) != len(df):
  1502→          print(f"警告：基准数据长度({len(benchmark_close)})与策略数据长度({len(df)})不一致")
  1503→          # 这里简单处理，取相同长度的数据
  1504→          min_len = min(len(benchmark_close), len(df))
  1505→          benchmark_close = benchmark_close.iloc[:min_len]
  1506→          benchmark_net_value = benchmark_net_value.iloc[:min_len]
  1507→          df = df.iloc[:min_len]
  1508→

  1511→          current_price = df['close'].iloc[i]

  1513→          sell_signal = df['sellxh'].iloc[i] if i < len(df) else 0
  1514→          
  1515→          # 记录交易前的状态
  1516→          pre_open_count = open_count
  1517→          pre_close_count = close_count
  1518→          pre_win_count = win_count
  1519→

  1521→          if buy_signal == 1 and not hold:
  1522→              # 计算可买整百股数量
  1523→              max_shares = int(rest_cash / current_price / 100) * 100
  1524→              if max_shares > 0:
  1525→                  # 计算手续费
  1526→                  trade_amount = max_shares * current_price
  1527→                  fee = trade_amount * fee_rate
  1528→                  
  1529→                  # 执行买入
  1530→                  position = max_shares
  1531→                  rest_cash = rest_cash - trade_amount - fee
  1532→                  hold = True
  1533→                  open_count += 1
  1534→                  
  1535→                  # 记录交易
  1536→                  trade_records.append({
  1537→                      'date': df.index[i],
  1538→                      'type': 'buy',
  1539→                      'price': current_price,
  1540→                      'shares': position,
  1541→                      'fee': fee
  1542→                  })
  1543→          
  1544→          # 死叉卖出信号
  1545→          elif sell_signal == 1 and hold:
  1546→              # 计算卖出金额
  1547→              trade_amount = position * current_price
  1548→              fee = trade_amount * fee_rate
  1549→              
  1550→              # 执行卖出
  1551→              rest_cash = rest_cash + trade_amount - fee
  1552→              position = 0
  1553→              hold = False
  1554→              close_count += 1
  1555→

  1558→                  buy_price = trade_records[-1]['price']
  1559→                  if current_price > buy_price:
  1560→                      win_count += 1
  1561→              
  1562→              # 记录交易
  1563→              trade_records.append({
  1564→                  'date': df.index[i],
  1565→                  'type': 'sell',
  1566→                  'price': current_price,
  1567→                  'shares': position,
  1568→                  'fee': fee
  1569→              })
  1570→          
  1571→          # 计算当日市值和净值
  1572→          if hold:
  1573→              daily_value = rest_cash + position * current_price
  1574→          else:
  1575→              daily_value = rest_cash
  1576→
  1577→          daily_net_value = daily_value / initial_capital
  1578→
  1579→          # 计算胜率（到当前日期为止）
  1580→          total_trades_to_date = open_count + close_count
  1581→          win_rate_to_date = (win_count / total_trades_to_date * 100) if total_trades_to_date > 0 else 0
  1582→
  1583→          # 计算年化收益率（到当前日期为止）
  1584→          if i > 0:
  1585→              total_return_to_date = daily_net_value - 1
  1586→              trading_days_to_date = i + 1
  1587→              annual_return_to_date = (1 + total_return_to_date) ** (252 / trading_days_to_date) - 1 if trading_days_to_date > 0 else 0
  1588→          else:
  1589→              annual_return_to_date = 0
  1590→
  1591→          # 获取基准净值（使用真实的基准数据）
  1592→          if i < len(benchmark_net_value):
  1593→              current_benchmark_net_value = benchmark_net_value.iloc[i]
  1594→          else:
  1595→              current_benchmark_net_value = 1.0
  1596→
  1597→          # 计算基准年化收益率（到当前日期为止）
  1598→          if i > 0 and i < len(benchmark_net_value):
  1599→              benchmark_total_return_to_date = current_benchmark_net_value - 1
  1600→              benchmark_annual_return_to_date = (1 + benchmark_total_return_to_date) ** (252 / (i + 1)) - 1 if i > 0 else 0
  1601→          else:
  1602→              benchmark_annual_return_to_date = 0
  1603→
  1604→          # 计算贝塔值（到当前日期为止）
  1605→          if i > 1:
  1606→              # 计算策略收益率序列
  1607→              strategy_returns = []
  1608→              for j in range(i + 1):
  1609→                  if j == 0:
  1610→                      strategy_returns.append(0)
  1611→                  else:
  1612→                      prev_capital = daily_stats[j-1]['capital'] if j > 0 else initial_capital
  1613→                      curr_capital = daily_value if j == i else daily_stats[j]['capital']
  1614→                      strategy_return = (curr_capital / prev_capital) - 1
  1615→                      strategy_returns.append(strategy_return)
  1616→
  1617→              # 计算基准收益率序列
  1618→              benchmark_returns_to_date = benchmark_close.iloc[:i+1].pct_change().fillna(0).values
  1619→
  1620→              # 计算协方差和方差

  1622→                  cov_matrix = np.cov(strategy_returns, benchmark_returns_to_date)
  1623→                  beta = cov_matrix[0, 1] / cov_matrix[1, 1] if cov_matrix[1, 1] != 0 else 1.0
  1624→              else:
  1625→                  beta = 1.0
  1626→          else:
  1627→              beta = 1.0
  1628→
  1629→          # 收集每日统计数据
  1630→          daily_stats.append({
  1631→              'date': df.index[i],
  1632→              'capital': daily_value,
  1633→              'net_value': daily_net_value,

  1976→#### Q1: 同一个选股公式，`formula_process_mul_xg` 选股结果比客户端条件选股少？
  1977→**解决方案**: 检查 `count` 参数是否合理。`count` 必须满足公式计算所需的数据量。客户端条件选股使用了全部本地数据；`count` 太小会导致公式无法算出有效值。
  1978→- 提示：MACD 至少需要 100 根 K 线；UPN('3') 至少需要 5 根；如不确定，全市场回测用 `count=100` 起步。
