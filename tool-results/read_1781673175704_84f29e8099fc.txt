   128→   128→def calc_technical_indicators(group):
   129→   129→    group = group.sort_values('date').reset_index(drop=True)
   130→   130→    if len(group) < 5:
   131→   131→        return group

   135→   135→
   136→   136→    group['MA5'] = close.rolling(5, min_periods=1).mean()
   137→   137→    group['MA10'] = close.rolling(10, min_periods=1).mean()
   138→   138→    group['MA20'] = close.rolling(20, min_periods=1).mean()
   139→   139→
   140→   140→    ema12 = close.ewm(span=12, adjust=False).mean()
   141→   141→    ema26 = close.ewm(span=26, adjust=False).mean()
   142→   142→    dif = ema12 - ema26
   143→   143→    dea = dif.ewm(span=9, adjust=False).mean()
   144→   144→    group['DIF'] = dif
   145→   145→    group['DEA'] = dea
   146→   146→    group['MACD_BAR'] = 2 * (dif - dea)
   147→   147→
   148→   148→    group['BOLL_MID'] = close.rolling(20, min_periods=1).mean()
   149→   149→    boll_std = close.rolling(20, min_periods=1).std()
   150→   150→    group['BOLL_UP'] = group['BOLL_MID'] + 2 * boll_std
   151→   151→    return group

   236→   236→def calc_pressure_level(code, kline_data, latest_close):
   237→   237→    if code not in kline_data.index:
   238→   238→        return latest_close * 1.02, '估算'
   239→   239→    row = kline_data.loc[code]
   240→   240→    if isinstance(row, pd.DataFrame): row = row.iloc[0]
   241→   241→    candidates = []
   242→   242→    prev_high = float(row.get('high', 0))
   243→   243→    if prev_high > latest_close * 1.005: candidates.append((prev_high, '前日高点'))
   244→   244→    boll_up = row.get('BOLL_UP', np.nan)
   245→   245→    if pd.notna(boll_up) and float(boll_up) > latest_close * 1.005: candidates.append((float(boll_up), '布林上轨'))
   246→   246→    for ma_key in ['MA20', 'MA10', 'MA5']:
   247→   247→        ma_val = row.get(ma_key, np.nan)
   248→   248→        if pd.notna(ma_val) and float(ma_val) > latest_close * 1.003: candidates.append((float(ma_val), ma_key))
   249→   249→    if not candidates:
   250→   250→        zt_price = latest_close * 1.10
   251→   251→        code_str = str(code)
   252→   252→        if code_str.startswith(('688','689','300','301','302')): zt_price = latest_close * 1.20
   253→   253→        elif code_str.startswith(('8','4')): zt_price = latest_close * 1.30
   254→   254→        return round(zt_price, 2), '涨停价'
   255→   255→    candidates.sort(key=lambda x: abs(x[0] - latest_close))
   256→   256→    return round(candidates[0][0], 2), candidates[0][1]

   264→   264→def score_daban(df):
   265→   265→    pool = df[
   266→   266→        (df['是否涨停']) |
   267→   267→        (df['ZAF'] >= 7)
   268→   268→    ].copy()
   269→   269→    scores = pd.DataFrame(index=pool.index)

   271→   271→    scores['股票名称'] = pool['股票名称']
   272→   272→    lp = {idx: [] for idx in pool.index}
   273→   273→
   274→   274→    # (1) 封板强度(0-40) — fLianB已重定义为连板数,补充原fLianB作为封板强度系数
   275→   275→    #     涨停基础分10 + 封成比加成(0-20) + 封单额加成(0-10) + 封板强度系数加成(0-5)
   276→   276→    scores['封板强度'] = 0
   277→   277→    fc_b = pool['FCb']; fc_amo = pool['FCAmo']
   278→   278→    fcb_strength = pool.get('封板强度系数', pd.Series(0, index=pool.index))
   279→   279→    scores.loc[pool['是否涨停'], '封板强度'] += 10
   280→   280→    for idx in pool[pool['是否涨停']].index: lp[idx].append('涨停')
   281→   281→    scores.loc[fc_b >= 0.5, '封板强度'] += 20
   282→   282→    scores.loc[(fc_b >= 0.2) & (fc_b < 0.5), '封板强度'] += 14
   283→   283→    scores.loc[(fc_b >= 0.05) & (fc_b < 0.2), '封板强度'] += 8
   284→   284→    scores.loc[(fc_b > 0) & (fc_b < 0.05), '封板强度'] += 3
   285→   285→    for idx in pool[fc_b >= 0.2].index: lp[idx].append('封板稳固')
   286→   286→    scores.loc[fc_amo >= 30000, '封板强度'] += 10
   287→   287→    scores.loc[(fc_amo >= 10000) & (fc_amo < 30000), '封板强度'] += 8
   288→   288→    scores.loc[(fc_amo >= 5000) & (fc_amo < 10000), '封板强度'] += 5
   289→   289→    scores.loc[(fc_amo >= 1000) & (fc_amo < 5000), '封板强度'] += 2
   290→   290→    for idx in pool[fc_amo >= 10000].index: lp[idx].append('大封单')
